        :param rnn_size: Size of RNNs
        :return: Tuple (cell, initialize state)
        """
        lstm_layers = 2
        # TODO: Implement Function
        lstm = tf.contrib.rnn.BasicLSTMCell(rnn_size)
        cell = tf.contrib.rnn.MultiRNNCell([lstm] * lstm_layers)
